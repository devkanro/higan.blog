import React from 'react';

export function QAItem({ question, children }) {
  const normalizedAnswer = React.Children.toArray(children).reduce((acc, child) => {
    if (typeof child === 'string') {
      const trimmed = child.trim();
      if (trimmed) {
        acc.push(trimmed);
      }
      return acc;
    }

    if (React.isValidElement(child) && child.type === 'p') {
      acc.push(...React.Children.toArray(child.props.children));
      return acc;
    }

    acc.push(child);
    return acc;
  }, []);

  return (
    <div
      style={{
        border: '1px solid #e5e7eb',
        borderRadius: '12px',
        padding: '12px 16px',
        margin: '12px 0',
        backgroundColor: '#f9fafb',
        boxShadow: '0 1px 2px rgba(0,0,0,0.03)'
      }}
    >
      <p style={{ margin: '0 0 6px', fontWeight: 600, color: '#b45309' }}>❓ {question}</p>
      <div
        style={{
          margin: 0,
          display: 'flex',
          gap: '6px',
          flexWrap: 'wrap',
          alignItems: 'baseline'
        }}
      >
        <span style={{ fontWeight: 600 }}>💬</span>
        <span>{normalizedAnswer}</span>
      </div>
    </div>
  );
}

## 🏢 架空层整改方案

### 整改决议
经会议讨论，各楼栋架空层确定实施封闭，由开发商统一规划和确定各楼栋的功能主题，以提升社区公共空间的整体品质和使用效率。

### 整改周期
整改周期采取“尽力而为”的原则推进：在交付前保证最大努力施工，其余部分将在交付后继续推进，过程中保持整体规划方案不发生变更。

### 功能规划
各楼栋架空层功能安排记录如下：

| 楼栋 | 功能主题 | 用途说明 |
|------|----------|----------|
| **1#楼** | 🏢 共享办公 | 为居民提供灵活办公空间 |
| **2#楼** | 👴 年轮臻享 | 老年人休闲娱乐活动区域 |
| **3#楼** | 🧒 儿童工坊 | 儿童手工制作与学习空间 |
| **4#楼** | 🎥 露营剧场 | 多功能文艺表演与观影区 |
| **5#楼** | 🤝 共享活动 | 社区集体活动举办场所 |
| **6#楼** | 🏃 健身轻运动 | 室内健身器材与运动区域 |
| **7#楼** | ☕ 品质会客 | 高端接待与商务洽谈空间 |
| **9#楼** | 📦 快递自助 | 智能快递柜与收发服务站 |

## 🚪 东门改造方案

会议期间，开发商展示了两套东门改造示意图但未允许拍照。现场业主多数倾向第一方案，因此纪要仅记录该方案的文字描述：

- 东门两侧围墙整体抬升至 3 米，并以香槟金色不锈钢饰面处理，墙体两侧配置 Logo、水池景观与绿植。
- 中轴消防入口的闸门将改为更密集的金色门片，左侧同步布置景观墙与绿植。
- 右侧人行出入口的保安厅经过美化改造，将延伸覆盖至入口通道，闸机区域增加顶棚，并更换为更高的闸机。
- 入口右侧墙体与左侧保持 3 米高度，用同款香槟金不锈钢装饰，并配置 Logo、水景与绿植。
